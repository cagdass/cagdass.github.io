Sorting

// Numbers ascending

// Numbers descending
arr.sort((a, b) => b - a)

// Alphabetical
arr.sort((a, b) => a.localeCompare(b))

// Multi-field sort (amount desc, then name asc)
arr.sort((a, b) => {
  if (b.amount !== a.amount) return b.amount - a.amount
  return a.name.localeCompare(b.name)
})

// Sort objects by field
arr.sort((a, b) => a.timestamp - b.timestamp)

Binary Search

// Array must be sorted first!
function binarySearch(arr: number[], target: number): number {
  let lo = 0, hi = arr.length - 1

  while (lo <= hi) {
    const mid = Math.floor((lo + hi) / 2)
    if (arr[mid] === target) return mid
    else if (arr[mid] < target) lo = mid + 1
    else hi = mid - 1
  }

  return -1  // not found
}

Class Design Pattern (Full Template)

class Transaction {
  id: string
  fromAccount: string
  toAccount: string
  amount: number
  currency: string
  status: "pending" | "completed" | "failed"
  timestamp: number

  constructor(id, fromAccount, toAccount, amount, currency, status, timestamp) {
    this.id = id; this.fromAccount = fromAccount; this.toAccount = toAccount
    this.amount = amount; this.currency = currency
    this.status = status; this.timestamp = timestamp
  }
}

class Ledger {
  private transactions: Transaction[] = []
  private converter?: CurrencyConverter

  constructor(converter?: CurrencyConverter) {
    this.converter = converter
  }

  addTransaction(tx: Transaction): void { this.transactions.push(tx) }

  getTransaction(id: string): Transaction | null {
    return this.transactions.find(tx => tx.id === id) ?? null
  }

  getBalance(accountId: string): number {
    let balance = 0
    for (const tx of this.transactions) {
      if (tx.status !== "completed") continue
      if (tx.toAccount === accountId) balance += tx.amount
      if (tx.fromAccount === accountId) balance -= tx.amount
    }
    return balance
  }
}

Binary Search

// Built-in (array must be sorted)
int[] arr = {1, 3, 5, 7};
int idx = Arrays.binarySearch(arr, 5); // returns 2

// Manual implementation
int binarySearch(int[] arr, int target) {
  int lo = 0, hi = arr.length - 1;
  while (lo <= hi) {
    int mid = lo + (hi - lo) / 2;
    if (arr[mid] == target) return mid;
    else if (arr[mid] < target) lo = mid + 1;
    else hi = mid - 1;
  }
  return -1;
}

Class Design Pattern (Full Template)

class Transaction {
  String id, fromAccount, toAccount, currency, status;
  double amount;
  long timestamp;

  Transaction(String id, String from, String to, double amount,
               String currency, String status, long timestamp) {
    this.id = id; this.fromAccount = from; this.toAccount = to;
    this.amount = amount; this.currency = currency;
    this.status = status; this.timestamp = timestamp;
  }
}

class Ledger {
  private List<Transaction> transactions = new ArrayList<>();
  private CurrencyConverter converter;  // nullable

  Ledger() {}
  Ledger(CurrencyConverter converter) { this.converter = converter; }

  void addTransaction(Transaction tx) { transactions.add(tx); }

  Transaction getTransaction(String id) {
    return transactions.stream()
      .filter(tx -> tx.id.equals(id))
      .findFirst()
      .orElse(null);
  }

  double getBalance(String accountId) {
    double balance = 0;
    for (Transaction tx : transactions) {
      if (!tx.status.equals("completed")) continue;
      if (tx.toAccount.equals(accountId)) balance += tx.amount;
      if (tx.fromAccount.equals(accountId)) balance -= tx.amount;
    }
    return balance;
  }
}

Streams (Optional but Useful)

import java.util.stream.*;

// Filter + collect
List<Transaction> completed = transactions.stream()
  .filter(tx -> tx.status.equals("completed"))
  .collect(Collectors.toList());

// Sum
double total = transactions.stream()
  .mapToDouble(tx -> tx.amount)
  .sum();

// Sort + limit (top N)
transactions.stream()
  .sorted((a, b) -> Double.compare(b.amount, a.amount))
  .limit(n)
  .collect(Collectors.toList());
💡 Streams are optional — a clean for-loop is always acceptable and sometimes clearer under time pressure.
