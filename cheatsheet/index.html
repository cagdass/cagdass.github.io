Class / Interface / Extends — The Real Pattern

NEW ★
// 1. Interface (defines the shape — no implementation)
interface ContainerInterface {
  add(value: number): number
  delete(value: number): boolean
}

// 2. Abstract class (like interface but CAN have some implementation)
abstract class ContainerBase {
  abstract add(value: number): number       // must be implemented
  helper() { return "shared logic" }         // can be used as-is
}

// 3. Class implementing interface
class MyContainer implements ContainerInterface {
  private items: number[] = []
  add(value: number): number { ... }
  delete(value: number): boolean { ... }
}

// 4. Class extending abstract class (what CodeSignal uses!)
class IntegerContainer extends IntegerContainerInterface {
  private container: number[] = []

  constructor() {
    super()  // ALWAYS call super() first when extending
  }

  add(value: number): number {
    this.container.push(value)
    return this.container.length
  }

  delete(value: number): boolean {
    const index = this.container.indexOf(value)
    if (index === -1) return false
    this.container.splice(index, 1)  // remove 1 element at index
    return true
  }
}
⚠️ CodeSignal gives you the base class — just extend it and call super(). Don't touch the interface file.

Median Pattern

NEW ★
getMedian(): number | null {
  if (this.container.length === 0) return null

  // Clone before sorting! Don't mutate the original
  const sorted = [...this.container].sort((a, b) => a - b)
  const n = sorted.length

  // "leftmost of two middle" for even length
  const midIndex = Math.floor((n - 1) / 2)
  return sorted[midIndex]
}

// Standard median (average of two middle for even)
if (n % 2 === 1) return sorted[Math.floor(n / 2)]
return (sorted[n/2 - 1] + sorted[n/2]) / 2
⚠️ Always read the spec — "leftmost of two middle" vs "average" are different!

// Binary search (array must be sorted!)
function binarySearch(arr: number[], target: number) {
  let lo = 0, hi = arr.length - 1
  while (lo <= hi) {
    const mid = Math.floor((lo + hi) / 2)
    if (arr[mid] === target) return mid
    else if (arr[mid] < target) lo = mid + 1
    else hi = mid - 1
  }
  return -1
}

Class / Interface / Extends — The Real Pattern

NEW ★
// 1. Interface
interface ContainerInterface {
  int add(int value);
  boolean delete(int value);
}

// 2. Class implementing interface
class MyContainer implements ContainerInterface {
  private List<Integer> items = new ArrayList<>();

  public int add(int value) {
    items.add(value);
    return items.size();
  }

  public boolean delete(int value) {
    return items.remove(Integer.valueOf(value)); // ← Important! Not index
  }
}

// 3. Abstract class + extends (CodeSignal pattern)
abstract class ContainerBase {
  abstract int add(int value);
}

class IntegerContainer extends ContainerBase {
  private List<Integer> container = new ArrayList<>();

  public IntegerContainer() {
    super(); // call parent constructor first
  }

  public int add(int value) {
    container.add(value);
    return container.size();
  }
}
⚠️ list.remove(int) removes by INDEX. list.remove(Integer.valueOf(x)) removes by VALUE. Easy bug!

Median Pattern

NEW ★
public Integer getMedian() {
  if (container.isEmpty()) return null;

  // Clone and sort
  List<Integer> sorted = new ArrayList<>(container);
  Collections.sort(sorted);
  int n = sorted.size();

  // "leftmost of two middle" for even length
  int midIndex = (n - 1) / 2;
  return sorted.get(midIndex);
}

// Standard median (average for even)
if (n % 2 == 1) return sorted.get(n / 2);
return (sorted.get(n/2-1) + sorted.get(n/2)) / 2.0;

Class / Inheritance — The Full Pattern

NEW ★
# 1. Base class (like abstract class)
class ContainerBase:
    def add(self, value: int) -> int:
        raise NotImplementedError

# 2. Class extending base class (CodeSignal pattern)
class IntegerContainer(ContainerBase):
    def __init__(self):
        super().__init__()      # call parent constructor
        self.container = []    # instance variable — always self.

    def add(self, value: int) -> int:
        self.container.append(value)
        return len(self.container)

    def delete(self, value: int) -> bool:
        if value not in self.container:
            return False
        self.container.remove(value)   # removes first occurrence
        return True

# 3. Using it
c = IntegerContainer()
c.add(5)    # → 1
c.delete(5) # → True
⚠️ Every method needs self as the first parameter. self.x is how you access instance variables.

Median Pattern

NEW ★
def get_median(self):
    if not self.container:
        return None

    sorted_c = sorted(self.container)  # new list, original safe
    n = len(sorted_c)

    # "leftmost of two middle" for even
    mid = (n - 1) // 2
    return sorted_c[mid]

# Standard median (average of two middle)
if n % 2 == 1:
    return sorted_c[n // 2]
return (sorted_c[n//2 - 1] + sorted_c[n//2]) / 2
